    <!-- Script หลักของระบบ -->
    <script src="https://cdn.jsdelivr.net/npm/@tensorflow/tfjs@latest/dist/tf.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/@teachablemachine/image@latest/dist/teachablemachine-image.min.js"></script>
    <script type="text/javascript">
        const URL = "https://teachablemachine.withgoogle.com/models/sig1pdXOe/";

        let model, webcam, labelContainer, maxPredictions;

        async function init() {
            // ซ่อนข้อความ placeholder เมื่อกล้องเริ่มทำงาน
            const placeholder = document.getElementById("webcam-placeholder");
            if(placeholder) placeholder.remove();

            const modelURL = URL + "model.json";
            const metadataURL = URL + "metadata.json";

            model = await tmImage.load(modelURL, metadataURL);
            maxPredictions = model.getTotalClasses();

            const flip = true; 
            webcam = new tmImage.Webcam(200, 200, flip); 
            await webcam.setup(); 
            await webcam.play();
            window.requestAnimationFrame(loop);

            document.getElementById("webcam-container").appendChild(webcam.canvas);
            labelContainer = document.getElementById("label-container");
            
            // ล้างค่าเก่าก่อนสร้างโครงสร้างแถบแสดงผลใหม่
            labelContainer.innerHTML = "";
            for (let i = 0; i < maxPredictions; i++) {
                // สร้างโครงสร้างการแสดงผลแบบมี Progress Bar เพื่อความสวยงาม
                const itemDiv = document.createElement("div");
                itemDiv.className = "bg-slate-50 p-3 rounded-xl border border-slate-100";
                itemDiv.innerHTML = `
                    <div class="flex justify-between text-sm font-medium text-slate-700 mb-1">
                        <span class="class-name">Loading...</span>
                        <span class="class-prob">0%</span>
                    </div>
                    <div class="w-full bg-slate-200 rounded-full h-2.5">
                        <div class="bg-indigo-600 h-2.5 rounded-full progress-bar" style="width: 0%"></div>
                    </div>
                `;
                labelContainer.appendChild(itemDiv);
            }
        }

        async function loop() {
            webcam.update(); 
            await predict();
            window.requestAnimationFrame(loop);
        }

        async function predict() {
            const prediction = await model.predict(webcam.canvas);
            for (let i = 0; i < maxPredictions; i++) {
                const className = prediction[i].className;
                const probability = prediction[i].probability;
                const percentage = (probability * 100).toFixed(0) + "%";

                // อัปเดตข้อความชื่อคลาส, เปอร์เซ็นต์ และความยาวของ Progress Bar ตามจริง
                const targetItem = labelContainer.childNodes[i];
                if(targetItem) {
                    targetItem.querySelector('.class-name').innerText = className;
                    targetItem.querySelector('.class-prob').innerText = percentage;
                    targetItem.querySelector('.progress-bar').style.width = percentage;
                }
            }
        }
    </script>
